class Demo:
            def __init__(self,x,y):
                self.sid=x
                self.sname=y
            def display(self):
                print(self.sid,self.sname)
                
        id=101
        name="KLU"
        d1 = Demo(id,name) #d1 is an object of Demo Class
        d1.display()
        d2 = Demo(102,"KLEF") #d2 is an object of Demo Class
        d2.display()
