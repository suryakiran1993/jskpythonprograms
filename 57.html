class A:
            id=101
            name="KLU"
            def display1(self):
                print("In Class A Display Method")
                print(self.id,self.name)
        
        class B:
            gender="FEMALE"
            maritalstatus = True
            def display2(self):
                print("In Class B Display Method")
                print(self.gender,self.maritalstatus)
        
        class C(A,B):
            cgpa = 8.58
            backlogs = 0
            def display3(self):
                print("In Class C Display Method")
                print(self.cgpa,self.backlogs,self.id,self.name,self.gender,self.maritalstatus)
        
        a = A()
        a.display1()
        b = B()
        b.display2()
        c = C()
        c.display3()
        c.display1()
        c.display2()
