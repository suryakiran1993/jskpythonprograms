class Student:
        id=101
        name="KLU"

        def display(self):
            print(f"ID={self.id},Name={self.name}")
    
        def __str__(self):
            return "Name=",self.name
    
        def __int__(self):
            return self.id
    
    s = Student() # s is an object of type Student
    s.display()
    print(s.__str__())
    print(s.__int__())
